00123             // no need to continue search in one direction when own entry is
00124             // reached:

00126             if (b == x) b = n;

00128             if (a >= 0) {

00130                 if ((!entry->node.isUnspecified()) &&
00131                         specialCloserCondition(entry->node, destination, *ret))
00132                     ret = &(entry->node);
00133                 a--;
00134             }
00135             if (b < n) {

00137                 if ((!entry->node.isUnspecified()) &&
00138                         specialCloserCondition(entry->node, destination, *ret))
00139                     ret = &(entry->node);

00142         }
00143

00151                 if ((!entry->node.isUnspecified()) &&
00152                         specialCloserCondition(entry->node, destination, *ret))

00158                     ret = &(entry->node);
00159             }

00163     } else {
00164         // no optimization, return the first closer node found
00165         for (uint32_t y = 0; y < rows.size(); ++y) {
