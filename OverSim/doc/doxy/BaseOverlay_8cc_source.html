00877 void BaseOverlay::handleBaseOverlayMessage(BaseOverlayMessage* msg,
00878                                            const OverlayKey& destKey)
00879 {
00880     switch (msg->getType()) {
00881     case OVERLAYSIGNALING:
00882         handleUDPMessage(msg);
00883         return;
00884 
00885     case RPC: {
00886         // process rpc-messages
00887         BaseRpcMessage* rpcMsg = check_and_cast<BaseRpcMessage*>(msg);
00888 
00889         // TODO destKey -> overlayCtrlInfo
00890         internalHandleRpcMessage(rpcMsg);
00891         return;
00892     }
00893 
00894     case APPDATA: {
00895         //TODO use route messages? here: set transport type to ROUTE for "naked"
00896         // app messages
00897         OverlayCtrlInfo* overlayCtrlInfo = check_and_cast<OverlayCtrlInfo*>(msg->getControlInfo());
00898         overlayCtrlInfo->setTransportType(ROUTE_TRANSPORT);
00899 
00900         BaseAppDataMessage* baseAppDataMsg =
00901             check_and_cast<BaseAppDataMessage*>(msg);
00902         callDeliver(baseAppDataMsg, destKey);
00903         return;
00904     }
00905 
00906     case OVERLAYROUTE: {
00907         BaseRouteMessage* baseRouteMsg =
00908             check_and_cast<BaseRouteMessage*>(msg);
00909 
00910         // collect delay-value of completed hop
00911         if (collectPerHopDelay) {
00912             baseRouteMsg->setHopDelayArraySize(baseRouteMsg->
00913                                                getHopDelayArraySize() + 1);
00914             baseRouteMsg->setHopDelay(baseRouteMsg->getHopDelayArraySize() - 1,
00915                                       simTime() - baseRouteMsg->getHopStamp());
00916         }
00917 
00918         OverlayCtrlInfo* overlayCtrlInfo
00919             = check_and_cast<OverlayCtrlInfo*>(baseRouteMsg
00920                                                ->removeControlInfo());
00921         // set transport type
00922         overlayCtrlInfo->setTransportType(ROUTE_TRANSPORT);
00923 
00924         // source routing: save visited nodes, copy next hops
00925         std::vector<TransportAddress> sourceRoute;
00926         if ((baseRouteMsg->getNextHopsArraySize() > 0) ||
00927              (baseRouteMsg->getRoutingType() == RECURSIVE_SOURCE_ROUTING) ||
00928              recordRoute) {
00929             // store the TransportAddress of the sender in the visited list
00930             baseRouteMsg->setVisitedHopsArraySize(baseRouteMsg
00931                                           ->getVisitedHopsArraySize() + 1);
00932             baseRouteMsg->setVisitedHops(baseRouteMsg
00933                                           ->getVisitedHopsArraySize() - 1,
00934                                         overlayCtrlInfo->getLastHop());
00935 
00936             // remove nodes from next hops and copy them to sourceRoute
00937             if (baseRouteMsg->getNextHopsArraySize() > 0) {
00938                 sourceRoute.resize(baseRouteMsg->getNextHopsArraySize()- 1);
00939                 for (uint32_t i = 1; i < baseRouteMsg->getNextHopsArraySize();
00940                      ++i) {
00941                     sourceRoute[i - 1] = baseRouteMsg->getNextHops(i);
00942                 }
00943                 baseRouteMsg->setNextHopsArraySize(0);
00944             }
00945         }
00946 
00947         overlayCtrlInfo->setSrcNode(baseRouteMsg->getSrcNode());
00948 
00949         // decapsulate msg if node is sibling for destKey
00950         // or message is at its destination node
00951         bool err;
00952         if ((sourceRoute.size() == 0) &&
00953             (baseRouteMsg->getDestKey().isUnspecified() ||
00954              isSiblingFor(thisNode, baseRouteMsg->getDestKey(), 1, &err)
00955              /*&& !err*/)) {
00956             overlayCtrlInfo->setHopCount(baseRouteMsg->getHopCount());
00957             overlayCtrlInfo->setRoutingType(baseRouteMsg->getRoutingType());
00958 
00959             if (baseRouteMsg->getVisitedHopsArraySize() > 0) {
00960                 // recorded route available => add to srcNode
00961                 NodeHandle srcRoute(baseRouteMsg->getSrcNode().getKey(),
00962                                    baseRouteMsg->getVisitedHops(0));
00963 
00964                 for (uint32_t i = 0; i < baseRouteMsg->getVisitedHopsArraySize(); ++i) {
00965                     srcRoute.appendSourceRoute(baseRouteMsg->getVisitedHops(i));
00966                 }
00967 
00968                 overlayCtrlInfo->setSrcRoute(srcRoute);
00969             } else if (baseRouteMsg->getDestKey().isUnspecified()) {
00970                 // directly received (neither key routed nor source routed)
00971                 // TODO: does this happen for a BaseRouteMessage?
00972                 overlayCtrlInfo->setSrcRoute(
00973                         NodeHandle(baseRouteMsg->getSrcNode().getKey(),
00974                                    overlayCtrlInfo->getLastHop()));
00975             } else {
00976                 // route to key and no recorded route available
00977                 overlayCtrlInfo->setSrcRoute(baseRouteMsg->getSrcNode());
00978             }
00979 
00980             // copy visited nodes to control info
00981             overlayCtrlInfo->setVisitedHopsArraySize(
00982                     baseRouteMsg->getVisitedHopsArraySize());
00983 
00984             for (uint32_t i = 0; i < baseRouteMsg->getVisitedHopsArraySize();
00985                  ++i) {
00986                 overlayCtrlInfo->setVisitedHops(i,
00987                         baseRouteMsg->getVisitedHops(i));
00988             }
00989 
00990             BaseOverlayMessage* tmpMsg
00991                 = check_and_cast<BaseOverlayMessage*>(baseRouteMsg
00992                                                       ->decapsulate());
00993             tmpMsg->setControlInfo(overlayCtrlInfo);
00994 
00995             // delay between hops
00996             if (collectPerHopDelay) {
00997                 RECORD_STATS(
00998                     size_t i;
00999                     for (i = singleHopDelays.size();
01000                              i < baseRouteMsg->getHopDelayArraySize();) {
01001                         singleHopDelays.push_back(new HopDelayRecord[++i]);
01002                     }
01003 
01004                     i = baseRouteMsg->getHopDelayArraySize() - 1;
01005                     HopDelayRecord* hdr = singleHopDelays[i];
01006 
01007                     for (size_t j = 0; j <= i; ++j) {
01008                         hdr[j].count++;
01009                         hdr[j].val += baseRouteMsg->getHopDelay(j);
01010                     }
01011                 );
01012             }
01013 
01014             // handle encapsulated message at destination node
01015             if (((baseRouteMsg->getRoutingType() == ITERATIVE_ROUTING)
01016                     || (baseRouteMsg->getRoutingType() == EXHAUSTIVE_ITERATIVE_ROUTING)
01017             )
01018                     || recursiveRoutingHook(thisNode, baseRouteMsg)) {
01019                 handleBaseOverlayMessage(tmpMsg, baseRouteMsg->getDestKey());
01020                 delete baseRouteMsg;
01021             }
01022             return;
01023         } else {
01024             // forward msg if this node is not responsible for the key
01025             baseRouteMsg->setControlInfo(overlayCtrlInfo);
01026 
01027             // if this node is malicious drop the message
01028             if (isMalicious() && dropRouteMessageAttack) {
01029                 EV << "[BaseOverlay::handleBaseOverlayMessage() @ " << thisNode.getIp()
01030                 << " (" << thisNode.getKey().toString(16) << ")]\n"
01031                 << "    BaseRouteMessage gets dropped because this node is malicious"
01032                 << endl;
01033                 //std::cout << "malicious!" << std::endl;
01034                 RECORD_STATS(numDropped++;
01035                              bytesDropped += baseRouteMsg->getByteLength());
01036                 delete baseRouteMsg;
01037                 return;
01038             }
01039 
01040             sendToKey(baseRouteMsg->getDestKey(), baseRouteMsg, 1, sourceRoute);
01041             return;
01042         }
01043         break;
01044     }
01045 
01046     default:
01047         EV << "[BaseOverlay::handleBaseOverlayMessage() @ " << thisNode.getIp()
01048            << " (" << thisNode.getKey().toString(16) << ")]\n"
01049            << "    Received unknown message from UDP of type " << msg->getName()
01050            << endl;
01051         break;
01052     }
01053 }

01411 void BaseOverlay::sendToKey(const OverlayKey& key, BaseOverlayMessage* msg,
01412                             int numSiblings,
01413                             const std::vector<TransportAddress>& sourceRoute,
01414                             RoutingType routingType)
01415 {
01416     BaseRouteMessage* routeMsg = NULL;
01417 
01418     if (routingType == DEFAULT_ROUTING) routingType = defaultRoutingType;
01419 
01420     if (debugOutput) {
01421         EV << "[BaseOverlay::sendToKey() @ " << thisNode.getIp()
01422         << " (" << thisNode.getKey().toString(16) << ")]\n"
01423         << "    Sending " << msg <<  " to " << key
01424         << endl;
01425     }
01426 
01427     if (key.isUnspecified() &&
01428         !(sourceRoute.size() && !sourceRoute[0].isUnspecified()))
01429         throw cRuntimeError("BaseOverlay::sendToKey(): "
01430                             "unspecified destination address and key!");
01431 
01432     if (msg->getType() != OVERLAYROUTE) {
01433         assert(!msg->getControlInfo());
01434         routeMsg = new BaseRouteMessage("BaseRouteMessage");
01435         routeMsg->setType(OVERLAYROUTE);
01436         routeMsg->setRoutingType(routingType);
01437         routeMsg->setDestKey(key);
01438         routeMsg->setSrcNode(thisNode);
01439         routeMsg->setStatType(msg->getStatType());
01440         // copy the name of the inner message
01441         routeMsg->setName(msg->getName());
01442         routeMsg->setBitLength(BASEROUTE_L(routeMsg));
01443         routeMsg->encapsulate(msg);
01444 
01445         OverlayCtrlInfo* routeCtrlInfo = new OverlayCtrlInfo;
01446         routeCtrlInfo->setLastHop(thisNode);
01447         routeCtrlInfo->setTransportType(ROUTE_TRANSPORT);
01448         routeCtrlInfo->setRoutingType(routingType);
01449         routeMsg->setControlInfo(routeCtrlInfo);
01450 
01451         //message marked as not already forwarded to tier1
01452         routeMsg->setContextPointer(NULL);
01453     } else {
01454         routeMsg = check_and_cast<BaseRouteMessage*>(msg);
01455         routingType = static_cast<RoutingType>(routeMsg->getRoutingType());
01456     }
01457 
01458     // set timestamp for next hop
01459     if (collectPerHopDelay) {
01460         routeMsg->setHopStamp(simTime());
01461     }
01462 
01463     if (sourceRoute.size() && !sourceRoute[0].isUnspecified()) {
01464         // send msg to nextHop if specified (used for e.g. join rpcs)
01465         OverlayCtrlInfo* ctrlInfo = check_and_cast<OverlayCtrlInfo*>
01466             (routeMsg->getControlInfo());
01467         ctrlInfo->setTransportType(UDP_TRANSPORT);
01468         assert(routeMsg->getNextHopsArraySize() == 0);
01469         routeMsg->setNextHopsArraySize(sourceRoute.size());
01470         for (uint32_t i = 0; i < sourceRoute.size(); ++i)
01471             routeMsg->setNextHops(i, sourceRoute[i]);
01472         if (recursiveRoutingHook(sourceRoute[0], routeMsg)) { //test
01473             sendRouteMessage(sourceRoute[0], routeMsg, routeMsgAcks);
01474         }
01475         return;
01476     }
01477 
01478     if ((routingType == ITERATIVE_ROUTING)
01479         || (routingType == EXHAUSTIVE_ITERATIVE_ROUTING)
01480         ) {
01481 
01482         // create lookup and sent to key
01483         AbstractLookup* lookup = createLookup(routingType, routeMsg, NULL,
01484                                     (routeMsg->getStatType() == APP_DATA_STAT));
01485         lookup->lookup(routeMsg->getDestKey(), numSiblings, hopCountMax,
01486                        0, new SendToKeyListener(this, routeMsg));
01487     } else  {
01488         // recursive routing
01489         NodeVector* nextHops = findNode(routeMsg->getDestKey(),
01490                                         recNumRedundantNodes,
01491                                         numSiblings, routeMsg);
01492 
01493         if (nextHops->size() == 0) {
01494             EV << "[BaseOverlay::sendToKey() @ " << thisNode.getIp()
01495                << " (" << thisNode.getKey().toString(16) << ")]\n"
01496                << "    FindNode() returned NULL - dropping message"
01497                << endl;
01498             //std::cout << simTime() << " " << thisNode.getIp() << " " << state
01499             //          << " FindNode() returned NULL - dropping message "
01500             //          << routeMsg->getName() << " from "
01501             //          << routeMsg->getSrcNode() << std::endl;
01502 
01503             // statistics
01504             RECORD_STATS(numDropped++; bytesDropped += routeMsg->getByteLength());
01505             delete routeMsg;
01506         } else {
01507             // delete message if the hop count maximum is exceeded
01508             if (routeMsg->getHopCount() >= hopCountMax) {
01509 
01510                 EV << "[BaseOverlay::sendToKey() @ " << thisNode.getIp()
01511                    << " (" << thisNode.getKey().toString(16) << ")]\n"
01512                    << "    Discards " << routeMsg->getName() << " from "
01513                    << routeMsg->getSrcNode().getIp() << "\n"
01514                    << "    The hop count maximum has been exceeded ("
01515                    << routeMsg->getHopCount() << ">="
01516                    << hopCountMax << ")"
01517                    << endl;
01518                 //std::cout << "[BaseOverlay::sendToKey() @ " << thisNode.getIp()
01519                 //          << " (" << thisNode.getKey().toString(16) << ")]\n"
01520                 //          << "    Discards " << routeMsg->getName() << " from "
01521                 //          << routeMsg->getSrcNode().getIp() << "\n"
01522                 //          << "    The hop count maximum has been exceeded ("
01523                 //          << routeMsg->getHopCount() << ">="
01524                 //          << hopCountMax << ")"
01525                 //          << std::endl;
01526 
01527                 // statistics
01528                 RECORD_STATS(numDropped++;
01529                              bytesDropped += routeMsg->getByteLength());
01530                 delete routeMsg;
01531                 delete nextHops;
01532                 return;
01533             }
01534 
01535             OverlayCtrlInfo* overlayCtrlInfo =
01536                 dynamic_cast<OverlayCtrlInfo*>(routeMsg->getControlInfo());
01537             assert(overlayCtrlInfo);
01538 
01539             // check and choose nextHop candidate
01540             NodeHandle* nextHop = NULL;
01541             bool err, isSibling;
01542             isSibling = isSiblingFor(thisNode, routeMsg->getDestKey(),
01543                                      numSiblings, &err);
01544 
01545             // if route is recorded we can do a real loop detection
01546             std::set<TransportAddress> visitedHops;
01547             for (uint32_t i = 0; i < routeMsg->getVisitedHopsArraySize(); ++i) {
01548                 visitedHops.insert(routeMsg->getVisitedHops(i));
01549             }
01550 
01551             for (uint32_t index = 0; nextHop == NULL && nextHops->size() > index;
01552                  ++index) {
01553                 nextHop = &((*nextHops)[index]);
01554                 // loop detection
01555                 if (((overlayCtrlInfo->getLastHop() == *nextHop) &&
01556                      (*nextHop != thisNode)) ||
01557                      (visitedHops.find(*nextHop) != visitedHops.end()) ||
01558                      // do not forward msg to source node
01559                     ((*nextHop == routeMsg->getSrcNode()) &&
01560                      (thisNode != routeMsg->getSrcNode())) ||
01561                      // nextHop is thisNode, but isSiblingFor() is false
01562                     ((*nextHop == thisNode) && (!isSibling))) {
01563                     nextHop = NULL;
01564                 }
01565             }
01566 
01567             if (nextHop == NULL) {
01568                 if (!checkFindNode(routeMsg)) {
01569                     EV << "[BaseOverlay::sendToKey() @ " << thisNode.getIp()
01570                        << " (" << thisNode.getKey().toString(16) << ")]\n"
01571                        << "    Discards " << routeMsg->getName() << " from "
01572                        << routeMsg->getSrcNode().getIp() << "\n"
01573                        << "    No useful nextHop found!"
01574                        << endl;
01575                     //std::cout << thisNode.getIp() << " packet from "
01576                     //          << routeMsg->getSrcNode().getIp()
01577                     //          << " dropped: " << routeMsg
01578                     //          << " " << state << std::endl;
01579                     RECORD_STATS(numDropped++;
01580                                  bytesDropped += routeMsg->getByteLength());
01581                 }
01582                 delete routeMsg;
01583                 delete nextHops;
01584                 return;
01585             }
01586 
01587             assert(!nextHop->isUnspecified());
01588 
01589             // callForward to app
01590             if (useCommonAPIforward &&
01591                 dynamic_cast<BaseAppDataMessage*>(
01592                         routeMsg->getEncapsulatedPacket()) &&
01593                 routeMsg->getContextPointer() == NULL) {
01594                 callForward(routeMsg->getDestKey(), routeMsg, *nextHop);
01595                 delete nextHops;
01596                 return;
01597             }
01598             //message marked as not already forwarded
01599             routeMsg->setContextPointer(NULL);
01600 
01601             // is this node responsible?
01602             if (*nextHop == thisNode) {
01603                 if (isSibling && !err) {
01604                     //EV << "[BaseOverlay::sendToKey() @ " << thisNode.getIp()
01605                     //   << " (" << thisNode.getKey().toString(16) << ")]\n"
01606                     //   << "    Forwards msg for key " << routeMsg->getDestKey() "\n"
01607                     //   << "    to node " << (*nextHops)[0]
01608                     //   << endl;
01609                     delete nextHops;
01610                     assert(routeMsg->getControlInfo());
01611                     handleBaseOverlayMessage(routeMsg, key);
01612                     return;
01613                 } else {
01614                     throw cRuntimeError("isSiblingsFor() is true with an "
01615                                         "error: Erroneous method "
01616                                         "isSiblingFor()!");
01617                 }
01618             }
01619             // else forward msg if this node is not responsible for the key
01620             overlayCtrlInfo->setHopCount(routeMsg->getHopCount());
01621             if (recursiveRoutingHook(*nextHop, routeMsg)) {
01622                 sendRouteMessage(*nextHop, routeMsg, routeMsgAcks);
01623             }
01624         }
01625         delete nextHops;
01626     }
01627 }
