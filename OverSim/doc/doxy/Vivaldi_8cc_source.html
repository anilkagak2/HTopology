00110 double Vivaldi::calcError(const simtime_t& rtt, double dist, double weight)
00111 {
00112     double relErr = 0;
00113     if (rtt != 0) {
00114         //eSample computes the relative error for this sample
00115         relErr = fabs(dist - rtt) / rtt;
00116     }
00117     // update weighted moving average of local error
00118     return (relErr * errorC * weight) +
00119            ownCoords->getError() * (1 - errorC * weight);
