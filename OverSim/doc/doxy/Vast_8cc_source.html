00625 void Vast::processPingTimer()
00626 {
00627     bool abnormalLeave = false;
00628     bool boundaryLeave = false;
00629     std::set<NodeHandle> removeSet;
00630     for(SiteMap::iterator itSites = Sites.begin(); itSites != Sites.end(); ++itSites) {
00631         if(itSites->second->tstamp < 0.0) { // node is dropped cause no pong has been received see below
00632             abnormalLeave = true;
00633             if(!(itSites->second->type & NEIGHBOR)) boundaryLeave = true;
00634             itSites->second->type = UNDEF;
00635             removeSet.insert( itSites->first );
00636         }
00637         else if(itSites->second->tstamp < simTime() - pingTimeout) { // node showed no activity for some time request pong and mark it to be dropped next time
00638             VastMessage *vastMsg = new VastMessage("PING");
00639             vastMsg->setCommand(PING);
00640             vastMsg->setBitLength(VAST_L(vastMsg));
00641             sendMessage(vastMsg, itSites->second->addr);
00642             itSites->second->tstamp = -1.0;
00643         }
00644     }
00645     if(abnormalLeave) {
00646         synchronizeApp();
00647         for( std::set<NodeHandle>::iterator it = removeSet.begin(); it != removeSet.end(); ++it) {
00648             removeNode( *it );
00649         }
00650         // removeNeighbors();
00651         if(boundaryLeave) {
00652             for(SiteMap::iterator itSites = Sites.begin(); itSites != Sites.end(); ++itSites) {
00653                 if(itSites->second->type & BOUNDARY) {
00654                     VastMessage *vastMsg = new VastMessage("ENCLOSING_NEIGHBORS_REQUEST");
00655                     vastMsg->setCommand(ENCLOSING_NEIGHBORS_REQUEST);
00656                     vastMsg->setBitLength(VAST_L(vastMsg));
00657                     sendMessage(vastMsg, itSites->second->addr);
00658                 }
00659             }
00660         }
00661         //buildVoronoi();
00662         //removeNeighbors(); should be superfluous
00663     }
00664 }

00783 void Vast::handleJoinRequest(VastMessage *vastMsg)
00784 {
00785     Site *forwardSite = NULL;
00786     // start with this node
00787     double min_dist = thisSite.coord.distanceSqr(vastMsg->getPos());
00788     forwardSite = &thisSite;
00789     // iterate through all neighbors
00790     for(SiteMap::iterator itSites = Sites.begin(); itSites != Sites.end(); ++itSites) {
00791         // dont forward to nodes which are still joining
00792         if(itSites->second->coord.distanceSqr(vastMsg->getPos()) < min_dist && itSites->second->neighborCount >= 0) {
00793             min_dist = itSites->second->coord.distanceSqr(vastMsg->getPos());
00794             forwardSite = itSites->second;
00795         }
00796     }
00797     // do nothing and let node retry with new position if current position is illegal
00798     if(min_dist == 0.0) {
00799         delete vastMsg;
00800     }
00801     else {
00802         // send an acknowledge or forward request if any of our neighbors is closer to joining node
00803         if(forwardSite->type & THIS) {
00804             VastListMessage *vastListMsg = new VastListMessage("JOIN_ACKNOWLEDGE");
00805             vastListMsg->setCommand(JOIN_ACKNOWLEDGE);
00806             // fill neighbors list
00807             vastListMsg->setNeighborNodeArraySize(Sites.size());
00808             vastListMsg->setNeighborPosArraySize(Sites.size());
00809             int i = 0;
00810             for(SiteMap::iterator itSites = Sites.begin(); itSites != Sites.end(); ++itSites) {
00811                 vastListMsg->setNeighborNode(i, itSites->second->addr);
00812                 vastListMsg->setNeighborPos(i, itSites->second->coord);
00813                 ++i;
00814             }
00815 
00816             vastListMsg->setBitLength(VASTLIST_L(vastListMsg));
00817             sendMessage(vastListMsg, vastMsg->getSourceNode());
00818 
00819             // add node to list to propagte its position early
00820             // nieghborCount is set to -1 to indicate node is still joining
00821             addNode(vastMsg->getPos(), vastMsg->getSourceNode(), -1);
00822             // update voronoi with new neighbors
00823             buildVoronoi();
00824             synchronizeApp();
00825             // removeNeighbors();
00826             delete vastMsg;
00827         }
00828         else {
00829             sendMessage(vastMsg, forwardSite->addr);
00830         }
00831     }

01056        Sites.find(vastMoveMsg->getSourceNode()) != Sites.end() &&

01059         sgcMsg->setNeighborPosition(addSize, vastMoveMsg->getNewPos());

01064     sgcMsg->setAddNeighborArraySize(addSize);
