00109 {
00110     std::vector<PastryExtendedNode>::iterator it;
00111 
00112     bool nodeAlreadyInVector = false; // was the node already in the list?
00113     bool nodeValueWasChanged = false;  // true if the list was changed, false if the rtt was too big
00114     // look for node in the set, if it's there and the value was changed, erase it (since the position is no longer valid)
00115     for (it = neighbors.begin(); it != neighbors.end(); it++) {
00116         if (!it->node.isUnspecified() && it->node == node) {
00117             if (prox == SimTime::getMaxTime() || it->rtt == prox) return false; // nothing to do!
00118             neighbors.erase(it);
00119             nodeAlreadyInVector = true;
00120             break;
00121         }
00122     }
00123     // look for the correct position for the node
00124     for (it = neighbors.begin(); it != neighbors.end(); it++) {
00125         if (it->node.isUnspecified() || (it->rtt > prox)) {
00126             nodeValueWasChanged = true;
00127             break;
00128         }
00129     }
00130     neighbors.insert(it, PastryExtendedNode(node, prox)); // insert the entry there
00131     if (!nodeAlreadyInVector) neighbors.pop_back(); // if a new entry was inserted, erase the last entry
00132     return !nodeAlreadyInVector && nodeValueWasChanged; // return whether a new entry was added
