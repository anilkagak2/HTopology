01076 std::pair<simtime_t, simtime_t> NeighborCache::getMeanVarRtt(const TransportAddress &node,
01077                                                              bool returnVar)

01079     if (neighborCache.count(node) == 0) {
01080         throw cRuntimeError("NeighborCache.cc: getMeanVarRtt was asked for"

01082     }
01083 
01084     uint16_t size = neighborCache[node].lastRtts.size();
01085     if (size == 0) return std::make_pair(-1.0,-1.0);
01086 
01087     simtime_t rttSum = 0;
01088     for (int i = 0; i < size; i++){
01089         rttSum += neighborCache[node].lastRtts[i];
01090     }
01091     simtime_t meanRtt = rttSum / size;
01092     if (!returnVar) {
01093         return std::make_pair(meanRtt, -1.0);

01095     if (size == 1) {
01096         return std::make_pair(meanRtt, 0.0);
01097     }
01098 
01099     double sum = 0.0;
01100     for (int i = 0; i < size; i++){
01101         simtime_t tempRtt = neighborCache[node].lastRtts.at(i) - meanRtt;
01102         sum += (SIMTIME_DBL(tempRtt) * SIMTIME_DBL(tempRtt));

01104
