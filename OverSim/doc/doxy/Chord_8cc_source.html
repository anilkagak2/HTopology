00444     if ((predecessorNode.isUnspecified()) && (node == thisNode)) {

00456

00462     NodeHandle curNode;
00463 
00464     for (int i = -1; i < (int)successorList->getSize();
00465          i++, prevNode = curNode) {
00466

00470             curNode = successorList->getSuccessor(i);
00471         }
00472 
00473         if (node == curNode) {
00474             // is the message destined for curNode?

00480                     *err = true;

00671     }

00674     throw cRuntimeError("Error in Chord::closestPreceedingNode()!");

00919 void Chord::rpcJoin(JoinCall* joinCall)
00920 {
00921     NodeHandle requestor = joinCall->getSrcNode();
00922 
00923     // compile successor list
00924     JoinResponse* joinResponse =
00925         new JoinResponse("JoinResponse");
00926 
00927     int sucNum = successorList->getSize();
00928     joinResponse->setSucNum(sucNum);
00929     joinResponse->setSucNodeArraySize(sucNum);
00930 
00931     for (int k = 0; k < sucNum; k++) {
00932         joinResponse->setSucNode(k, successorList->getSuccessor(k));
00933     }
00934 
00935     // sent our predecessor as hint to the joining node
00936     if (predecessorNode.isUnspecified() && successorList->isEmpty()) {
00937         // we are the only node in the ring
00938         joinResponse->setPreNode(thisNode);
00939     } else {
00940         joinResponse->setPreNode(predecessorNode);
00941     }
00942 
00943     joinResponse->setBitLength(JOINRESPONSE_L(joinResponse));
00944 
00945     sendRpcResponse(joinCall, joinResponse);
00946 
00947     if (aggressiveJoinMode) {
00948         // aggressiveJoinMode differs from standard join operations:
00949         // 1. set our predecessor pointer to the joining node
00950         // 2. send our old predecessor as hint in JoinResponse msgs
00951         // 3. send a NEWSUCCESSORHINT to our old predecessor to update
00952         //    its successor pointer
00953 
00954         // send NEWSUCCESSORHINT to our old predecessor
00955 
00956         if (!predecessorNode.isUnspecified()) {
00957             NewSuccessorHintMessage* newSuccessorHintMsg =
00958                 new NewSuccessorHintMessage("NEWSUCCESSORHINT");
00959             newSuccessorHintMsg->setCommand(NEWSUCCESSORHINT);
00960 
00961             newSuccessorHintMsg->setSrcNode(thisNode);
00962             newSuccessorHintMsg->setPreNode(requestor);
00963             newSuccessorHintMsg->
00964             setBitLength(NEWSUCCESSORHINT_L(newSuccessorHintMsg));
00965 
00966             sendMessageToUDP(predecessorNode, newSuccessorHintMsg);
00967         }
00968 
00969         if (predecessorNode.isUnspecified() || (predecessorNode != requestor)) {
00970             // the requestor is our new predecessor
00971             NodeHandle oldPredecessor = predecessorNode;
00972             predecessorNode = requestor;
00973 
00974             // send update to application if we've got a new predecessor
00975             if (!oldPredecessor.isUnspecified()) {
00976                 callUpdate(oldPredecessor, false);
00977             }
00978             callUpdate(predecessorNode, true);
00979 
00980         }
00981     }
00982 
00983     // if we don't have a successor, the requestor is also our new successor
00984     if (successorList->isEmpty())
00985         successorList->addSuccessor(requestor);
00986 
00987     updateTooltip();
00988 }

00990 void Chord::handleRpcJoinResponse(JoinResponse* joinResponse)
00991 {
00992     // determine the numer of successor nodes to add
00993     int sucNum = successorListSize - 1;
00994

00996         sucNum = joinResponse->getSucNum();
00997     }
00998 
00999     // add successor getNode(s)
01000     for (int k = 0; k < sucNum; k++) {
01001         NodeHandle successor = joinResponse->getSucNode(k);
01002         successorList->addSuccessor(successor);
01003     }
01004 
01005     // the sender of this message is our new successor
01006     successorList->addSuccessor(joinResponse->getSrcNode());
01007 
01008     // in aggressiveJoinMode: use hint in JoinResponse
01009     // to set our new predecessor
01010     if (aggressiveJoinMode) {
01011         // it is possible that the joinResponse doesn't contain a valid
01012         // predecessor especially when merging two partitions
01013         if (!joinResponse->getPreNode().isUnspecified()) {
01014             if (!predecessorNode.isUnspecified()) {
01015

01025                         setBitLength(NEWSUCCESSORHINT_L(newSuccessorHintMsg));

01032             predecessorNode = joinResponse->getPreNode();
01033 
01034             if (!oldPredecessor.isUnspecified()
01035                 && !joinResponse->getPreNode().isUnspecified()
01036                 && oldPredecessor != joinResponse->getPreNode()) {

01048     cancelEvent(stabilize_timer);

01123         (!call->getFailed().isUnspecified() &&
