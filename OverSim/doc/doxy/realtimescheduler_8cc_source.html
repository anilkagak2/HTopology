00300 int RealtimeScheduler::receiveUntil(const timeval& targetTime)
00301 {
00302     // if there's more than 200ms to wait, wait in 100ms chunks
00303     // in order to keep UI responsiveness by invoking ev.idle()
00304     timeval curTime;
00305     gettimeofday(&curTime, NULL);
00306     while (targetTime.tv_sec-curTime.tv_sec >=2 ||
00307             timeval_diff_usec(targetTime, curTime) >= 200000) {
00308         if (receiveWithTimeout(100000)) { // 100ms
00309             if (ev.idle()) return -1;
00310             return 1;
00311         }
00312         if (ev.idle()) return -1;
00313         gettimeofday(&curTime, NULL);
00314     }
00315 
00316     // difference is now at most 100ms, do it at once
00317     long usec = timeval_diff_usec(targetTime, curTime);
00318     if (usec>0)
00319         if (receiveWithTimeout(usec)) {
00320             if (ev.idle()) return -1;
00321             return 1;
00322         }
00323     if (ev.idle()) return -1;
00324     return 0;
00325 }
