00090 void TCPExampleApp::handleTimerEvent(cMessage* msg)
00091 {
00092     if (msg == timerMsg) {
00093         // if the simulator is still busy creating the network,
00094         // let's wait a bit longer
00095         if (underlayConfigurator->isInInitPhase()) {
00096             scheduleAt(simTime() + sendPeriod, timerMsg);
00097             return;
00098         }
00099 
00100         // do the same, if the node is the only one alive
00101         if (globalNodeList->getNumNodes() == 1){
00102             scheduleAt(simTime() + sendPeriod, timerMsg);
00103             return;
00104         }
00105 
00106         // get the address of one of the other nodes
00107         TransportAddress* addr = globalNodeList->getRandomAliveNode();
00108         while ((addr != NULL) && (thisNode.getIp().equals(addr->getIp()))) {
00109             addr = globalNodeList->getRandomAliveNode();
00110         }
00111 
00112         assert(addr != NULL);
00113 
00114         // create a PING message
00115         TCPExampleMessage *TCPMsg = new TCPExampleMessage();
00116         TCPMsg->setType(TCPEXMSG_PING); // set the message type to PING
00117         TCPMsg->setSenderAddress(thisNode); // set the sender address to our own
00118         TCPMsg->setByteLength(100); // set the message length to 100 bytes
00119 
00120         RECORD_STATS(numSent++); // update statistics
00121 
00122         // connect and send message
00123         TransportAddress remoteAddress = TransportAddress(addr->getIp(), 24000);
00124         establishTcpConnection(remoteAddress);
00125         sendTcpData(TCPMsg, remoteAddress);
00126 
00127         // user output
00128         EV << thisNode.getIp() << ": Connecting to "<< addr->getIp()
00129            << " and sending PING."<< std::endl;
