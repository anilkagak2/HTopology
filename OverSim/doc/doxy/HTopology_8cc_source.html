01141 // 2) Main procedure for deciding the replacement for keyParent
01142 void HTopology::goAheadWithRestSelectionProcess(const OverlayKey& key) {
01143     if (leaveRequests.find(key) != leaveRequests.end()) {
01144         EV << "some spurious call to goAheadWithRestSelectionAlgo" << endl;
01145         return;
01146     }
01147 
01148     if (children.find(key) == children.end()) {
01149         EV << "replacement of the given node is not a responsibility of this node " << endl;
01150         return;
01151     }
01152 
01153     int noOfChildrenToAdd = children[key].getNodeVector().size();
01154     bool replacementDone = false;
01155     std::map<OverlayKey, int>& queryNodesSelectionAlgo = leaveRequests[key].queryNodesSelectionAlgo;
01156     std::map<OverlayKey, int>::iterator it=leaveRequests[key].queryNodesSelectionAlgo.begin();
01157 
01158     for (; it!=queryNodesSelectionAlgo.end(); ++it) {
01159         // What else is to be done?
01160         // Modify the existing next & prev pointers for the replacement's sibling
01161         // Modify the pointers for the siblings in the current overlay
01162         // And add nodeChildren as parent to the replacement node
01163         int capacity = (*it).second;
01164         // TODO actually one less -> leave itself
01165         if (capacity >= noOfChildrenToAdd) {
01166             replacementDone = true;
01167             break;
01168         }
01169     }
01170 
01171     if (replacementDone) {
01172         // 1) Notify the node about the result
01173         // Send the HLeaveOverlayResponse to the node, whose replacement has been found
01174         HLeaveOverlayResponse *rrpc = new HLeaveOverlayResponse();
01175         rrpc->setPermissionGranted(true);
01176         rrpc->setBitLength(HLEAVEOVERLAYRESPONSE_L(rrpc));
01177         sendRpcResponse(leaveRequests[key].mrpc, rrpc);
01178 
01179 
01180         // 2) Replace the child with this newly selected parent (children list)
01181         NodeVector newChildren = children[key].getNodeVector();
01182         NodeVector::iterator pos=newChildren.begin();
01183         for (; pos!=newChildren.end(); ++pos) {
01184             if ((*pos).getKey() == (*it).first) break;
01185         }
01186 
01187         NodeHandle newHandle = *pos;
01188         if (pos != newChildren.end()) newChildren.erase(pos);
01189         else {
01190             EV << "Is this even possible that the replacement node is not in the children set?" << endl;
01191         }
01192         HNode newNode = children[key];
01193         NodeHandle oldNode = newNode.getHandle();
01194         children.erase(key);
01195         newNode.setHandle(newHandle);
