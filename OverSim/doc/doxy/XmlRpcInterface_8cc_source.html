00961 bool XmlRpcInterface::readHeader(char* buf, uint32_t length)
00962 {
00963     // Read available data
00964     bool eof = false;
00965 
00966     state[curAppFd]._header.append(std::string(buf, length));
00967 
00968     if (length <= 0) {
00969         // Its only an error if we already have read some data
00970         if (state[curAppFd]._header.length() > 0)
00971             XmlRpcUtil::error("XmlRpcServerConnection::readHeader: error "
00972                 "while reading header.");
00973         return false;
00974     }
00975 
00976     XmlRpcUtil::log(4, "XmlRpcServerConnection::readHeader: read %d bytes.",
00977                     state[curAppFd]._header.length());
00978     char *hp = (char*)state[curAppFd]._header.c_str(); // Start of header
00979     char *ep = hp + state[curAppFd]._header.length(); // End of string
00980     char *bp = 0; // Start of body
00981     char *lp = 0; // Start of content-length value
00982     char *kp = 0; // Start of connection value
00983 
00984     for (char *cp = hp; (bp == 0) && (cp < ep); ++cp) {
00985         if ((ep - cp > 16) && (strncasecmp(cp, "Content-length: ", 16) == 0))
00986             lp = cp + 16;
00987         else if ((ep - cp > 12) && (strncasecmp(cp, "Connection: ", 12) == 0))
00988             kp = cp + 12;
00989         else if ((ep - cp > 4) && (strncmp(cp, "\r\n\r\n", 4) == 0))
00990             bp = cp + 4;
00991         else if ((ep - cp > 2) && (strncmp(cp, "\n\n", 2) == 0))
00992             bp = cp + 2;
00993     }
00994 
00995     // If we haven't gotten the entire header yet, return (keep reading)
00996     if (bp == 0) {
00997         // EOF in the middle of a request is an error, otherwise its ok
00998         if (eof) {
00999             XmlRpcUtil::log(4, "XmlRpcServerConnection::readHeader: EOF");
01000             if (state[curAppFd]._header.length() > 0)
01001                 XmlRpcUtil::error("XmlRpcServerConnection::readHeader: EOF while reading header");
01002             return false; // Either way we close the connection
01003         }
01004 
01005         return true; // Keep reading
01006     }
01007 
01008     // Decode content length
01009     if (lp == 0) {
01010         XmlRpcUtil::error("XmlRpcServerConnection::readHeader: No Content-length specified");
01011         return false; // We could try to figure it out by parsing as we read, but for now...
01012     }
01013 
01014     state[curAppFd]._contentLength = atoi(lp);
01015     if (state[curAppFd]._contentLength <= 0) {
01016         XmlRpcUtil::error(
01017                           "XmlRpcServerConnection::readHeader: Invalid Content-length specified (%d).",
01018                           state[curAppFd]._contentLength);
01019         return false;
01020     }
01021 
01022     XmlRpcUtil::log(
01023                      3,
01024                     "XmlRpcServerConnection::readHeader: specified content length is %d.",
01025                     state[curAppFd]._contentLength);
01026 
01027     // Otherwise copy non-header data to request buffer and set state to read request.
01028     state[curAppFd]._request = bp;
01029 
01030     // Parse out any interesting bits from the header (HTTP version, connection)
01031     state[curAppFd]._keepAlive = true;
01032     if (state[curAppFd]._header.find("HTTP/1.0") != std::string::npos) {
01033         if (kp == 0 || strncasecmp(kp, "keep-alive", 10) != 0)
01034             state[curAppFd]._keepAlive = false; // Default for HTTP 1.0 is to close the connection
01035     } else {
01036         if (kp != 0 && strncasecmp(kp, "close", 5) == 0)
01037             state[curAppFd]._keepAlive = false;
01038     }
01039     XmlRpcUtil::log(3, "KeepAlive: %d", state[curAppFd]._keepAlive);
01040 
01041     state[curAppFd]._header = "";
01042     state[curAppFd]._connectionState = READ_REQUEST;
01043     return true; // Continue monitoring this source
01044 }
