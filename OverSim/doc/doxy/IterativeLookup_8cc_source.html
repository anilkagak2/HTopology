00910 void IterativePathLookup::handleTimeout(BaseCallMessage* msg,
00911                                    const TransportAddress& dest, int rpcId)
00912 {
00913     if (finished)
00914         return;
00915 
00916     EV << "[IterativePathLookup::handleTimeout()]\n"
00917        << "    Timeout of RPC " << rpcId
00918        << endl;
00919 
00920     //std::cout << lookup->overlay->getThisNode() << ": Path timeout for node"
00921     //          << dest << endl;
00922 
00923     // For exhaustive-iterative remove dead nodes from nextHops vector
00924     // (which is also our results vector)
00925     if ((lookup->routingType == EXHAUSTIVE_ITERATIVE_ROUTING)
00926             && lookup->getDead(dest)) {
00927         LookupVector::iterator it = nextHops.findIterator(
00928                             (dynamic_cast<const NodeHandle&>(dest)).getKey());
00929         if (it != nextHops.end()) {
00930             nextHops.erase(it);
00931         }
00932     }
00933 
00934     std::map<TransportAddress, NodeHandle>::iterator oldPos;
00935     oldPos = oldNextHops.find(dest);
00936 
00937     // decrease pending rpcs
00938     pendingRpcs--;
00939 
00940     cPacket* findNodeExt = NULL;
00941     if (msg && msg->hasObject("findNodeExt")) {
00942         findNodeExt = static_cast<cPacket*>(
00943                 msg->removeObject("findNodeExt"));
00944     }
00945 
00946     if (simTime() > (lookup->startTime + LOOKUP_TIMEOUT)) {
00947         EV << "[IterativePathLookup::handleTimeout()]\n"
00948            << "    Iterative lookup path timed out!"
00949            << endl;
00950         delete findNodeExt;
00951         finished = true;
00952         success = false;
00953         return;
00954     }
00955 
00956     if (oldPos == oldNextHops.end() || (!lookup->config.failedNodeRpcs)) {
00957         sendNewRpcAfterTimeout(findNodeExt);
00958         delete findNodeExt;
00959     } else {
00960         if (oldPos->second.isUnspecified()) {
00961             // TODO: handleFailedNode should always be called for local
00962             // nodes, independent of config.failedNodeRpcs
00963             // Attention: currently this method is also called,
00964             // if a node responded and the path doesn't accept a message
00965             FindNodeCall* findNodeCall = dynamic_cast<FindNodeCall*>(msg);
00966             // answer was from local findNode()
00967 
00968             if (findNodeCall && lookup->overlay->handleFailedNode(dest)) {
00969                 NodeVector* retry = lookup->overlay->findNode(
00970                    findNodeCall->getLookupKey(), -1, lookup->numSiblings, msg);
00971 
00972                 for (NodeVector::iterator i = retry->begin(); i != retry->end(); i++) {

00975 
00976                 delete retry;

00981 
00982         } else {
00983             FailedNodeCall* call = new FailedNodeCall("FailedNodeCall");
00984             call->setFailedNode(dest);
00985             call->setBitLength(FAILEDNODECALL_L(call));
00986             if (findNodeExt) {

00988                 call->addBitLength(findNodeExt->getBitLength());

01057     // if strictParallelRpcs is true, limit concurrent in-flight requests

01060         num = min(num, lookup->config.parallelRpcs - pendingRpcs);

01074          ((num > 0) && (i < lookup->config.redundantNodes)
01075           && (it != nextHops.end())); it++, i++)  {
01076 
01077         // ignore nodes to which we've already sent an RPC
01078         if (it->alreadyUsed || lookup->getDead(it->handle)) continue;
