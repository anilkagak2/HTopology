00664 PastryStateMessage* BasePastry::createStateMessage(enum PastryStateMsgType type,
00665                                                    simtime_t timestamp,
00666                                                    int16_t row,
00667                                                    bool lastHop)
00668 {
00669     //checks
00670     if ((type & (PASTRY_STATE_JOIN | PASTRY_STATE_MINJOIN)) &&
00671         ((row == -1) || (timestamp != -1))) {
00672         throw cRuntimeError("Creating JOIN State without hopCount"
00673                             " or with timestamp!");
00674     } else if ((type & (PASTRY_STATE_ROUTINGROW)) && row == -1) {
00675         throw cRuntimeError("Creating ROUTINGROW State without row!");
00676     } else if ((type & (PASTRY_STATE_UPDATE)) &&
00677                 timestamp == -1) {
00678         throw cRuntimeError("Creating UPDATE/REPAIR State without timestamp!");
00679     } else if ((type & (PASTRY_STATE_LEAFSET | PASTRY_STATE_STD |
00680                         PASTRY_STATE_REPAIR  | PASTRY_STATE_JOINUPDATE)) &&
00681             ((row != -1) || (timestamp != -1))) {
00682             throw cRuntimeError("Creating STD/UPDATE State with row/hopCount"
00683                                 " or timestamp!");
00684     }
00685 
00686     std::string typeStr = cEnum::get("PastryStateMsgType")->getStringFor(type);
00687     typeStr.erase(0, 13);
00688 
00689     EV << "[BasePastry::sendStateTables() @ " << thisNode.getIp()
00690        << " (" << thisNode.getKey().toString(16) << ")]\n"
00691        << "    creating new state message (" << typeStr << ")" //TODO
00692        << endl;
00693 
00694     // create new state msg and set special fields for some types:
00695     PastryStateMessage* stateMsg =
00696         new PastryStateMessage((std::string("STATE (") + typeStr + ")").c_str());
00697 
00698     stateMsg->setTimestamp((timestamp == -1) ? simTime() : timestamp);
00699 
00700     if (type == PASTRY_STATE_JOIN || type == PASTRY_STATE_MINJOIN) {
00701         stateMsg->setRow(row);
00702         stateMsg->setLastHop(lastHop);
00703     }
00704 
00705     // fill in standard content:
00706     stateMsg->setStatType(MAINTENANCE_STAT);
00707     stateMsg->setPastryStateMsgType(type);
00708     stateMsg->setSender(thisNode);
00709 
00710     // the following part of the new join works on the assumption,
00711     // that the node routing the join message is close to the joining node
00712     // therefore it should be switched on together with the discovery algorithm
00713     if (type == PASTRY_STATE_MINJOIN) {
00714         //send just the needed row for new join protocol
00715         routingTable->dumpRowToMessage(stateMsg, row);
00716         if (lastHop) {
00717             leafSet->dumpToStateMessage(stateMsg);
00718         } else {
00719             stateMsg->setLeafSetArraySize(0);
00720         }
00721         if (row == 1) {
00722             neighborhoodSet->dumpToStateMessage(stateMsg);
00723         } else {
00724             stateMsg->setNeighborhoodSetArraySize(0);
00725         }
00726     } else if (type == PASTRY_STATE_LEAFSET) {
00727         leafSet->dumpToStateMessage(stateMsg);
00728     } else if (type == PASTRY_STATE_ROUTINGROW) {
00729         routingTable->dumpRowToMessage(stateMsg, row);
00730     } else {
00731         routingTable->dumpToStateMessage(stateMsg);
00732         leafSet->dumpToStateMessage(stateMsg);
00733         neighborhoodSet->dumpToStateMessage(stateMsg);
00734     }
00735 
00736     stateMsg->setBitLength(PASTRYSTATE_L(stateMsg));
00737 
00738     return stateMsg;
00739 }

00906 NodeVector* BasePastry::findNode(const OverlayKey& key,
00907                                  int numRedundantNodes,
00908                                  int numSiblings,
00909                                  BaseOverlayMessage* msg)
00910 {
00911     if ((numRedundantNodes > getMaxNumRedundantNodes()) ||
00912         (numSiblings > getMaxNumSiblings())) {
00913 
00914         opp_error("(Pastry::findNode()) numRedundantNodes or numSiblings "
00915                   "too big!");
00916     }
00917     RECORD_STATS(totalLookups++);
00918 
00919     NodeVector* nextHops = new NodeVector(numRedundantNodes);
00920 
00921     if (state != READY) {
00922         return nextHops;
00923     } else if (key.isUnspecified() || leafSet->isClosestNode(key)) {
00924         RECORD_STATS(responsibleLookups++);
00925         nextHops->add(thisNode);
00926     } else {
00927         const NodeHandle* next = &(leafSet->getDestinationNode(key));
00928 
00929         if (next->isUnspecified()) {
00930             next = &(routingTable->lookupNextHop(key));
00931             if (!next->isUnspecified()) {
00932                 RECORD_STATS(routingTableLookups++);
00933             }
00934         } else {
00935             RECORD_STATS(responsibleLookups++);
00936         }
00937 
00938         if (next->isUnspecified()) {
00939             RECORD_STATS(closerNodeLookups++);
00940             // call findCloserNode() on all state objects
00941             if (optimizeLookup) {
00942                 const NodeHandle* tmp;
00943                 next = &(routingTable->findCloserNode(key, true));
00944                 tmp = &(neighborhoodSet->findCloserNode(key, true));
00945 
00946                 if ((! tmp->isUnspecified()) &&
00947                     (leafSet->isCloser(*tmp, key, *next))) {
00948                     RECORD_STATS(closerNodeLookupsFromNeighborhood++);
00949                     next = tmp;
00950                 }
00951 
00952                 tmp = &(leafSet->findCloserNode(key, true));
00953                 if ((! tmp->isUnspecified()) &&
00954                     (leafSet->isCloser(*tmp, key, *next))) {
00955                     RECORD_STATS(closerNodeLookupsFromNeighborhood--);
00956                     next = tmp;
00957                 }
00958             } else {
00959                 next = &(routingTable->findCloserNode(key));
00960 
00961                 if (next->isUnspecified()) {
00962                     RECORD_STATS(closerNodeLookupsFromNeighborhood++);
00963                     next = &(neighborhoodSet->findCloserNode(key));
00964                 }
00965 
00966                 if (next->isUnspecified()) {
00967                     RECORD_STATS(closerNodeLookupsFromNeighborhood--);
00968                     next = &(leafSet->findCloserNode(key));
00969                 }
00970             }
00971         }
00972 
00973         iterativeJoinHook(msg, !next->isUnspecified());
00974 
00975         if (!next->isUnspecified()) {
00976             nextHops->add(*next);
00977         }
00978     }
00979 
00980     bool err;
00981 
00982     // if we're a sibling, return all numSiblings
00983     if ((numSiblings >= 0) && isSiblingFor(thisNode, key, numSiblings, &err)) {
00984         if (err == false) {
00985             delete nextHops;
00986             return  leafSet->createSiblingVector(key, numSiblings);
00987         }
00988     }
00989 
00990     if (/*(nextHops->size() > 0) &&*/ (numRedundantNodes > 1)) {
00991 
00992         //memleak... comp should be a ptr and deleted in NodeVector::~NodeVector()...
00993         //KeyDistanceComparator<KeyRingMetric>* comp =
00994         //    new KeyDistanceComparator<KeyRingMetric>( key );
00995 
00996         KeyDistanceComparator<KeyRingMetric> comp(key);
00997         //KeyDistanceComparator<KeyPrefixMetric> comp(key);
00998         NodeVector* additionalHops = new NodeVector( numRedundantNodes, &comp );
00999 
01000         routingTable->findCloserNodes(key, additionalHops);
01001         leafSet->findCloserNodes(key, additionalHops);
01002         neighborhoodSet->findCloserNodes(key, additionalHops);
01003 
01004         if (useRegularNextHop && (nextHops->size() > 0) &&
01005             (*additionalHops)[0] != (*nextHops)[0]) {
01006             for (uint32_t i = 0; i < additionalHops->size(); i++) {
01007                 if ((*additionalHops)[i] != (*nextHops)[0])
01008                     nextHops->push_back((*additionalHops)[i]);
01009             }
01010             delete additionalHops;
01011         } else {
01012             delete nextHops;
01013             return additionalHops;
01014         }
01015     }
01016     return nextHops;
01017 }
